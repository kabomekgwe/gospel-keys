## 3. DATA MODELS

### 3.1 Core Types

```typescript
// Note Event - Single MIDI note
interface NoteEvent {
  pitch: number;        // 0-127 (MIDI)
  start_time: number;   // seconds
  end_time: number;     // seconds
  velocity: number;     // 0-127
}

// Chord Event - Detected chord
interface ChordEvent {
  time: number;         // seconds
  duration: number;     // seconds
  chord: string;        // "Cmaj7", "Dm9", etc.
  confidence: number;   // 0-1
  root: string;         // "C", "D", etc.
  quality: string;      // "maj7", "m9", etc.
  bass_note?: string;   // For slash chords
  voicing_notes?: string[]; // [Phase 2] Actual notes
}

// Gospel Pattern - Detected progression pattern
interface GospelPattern {
  pattern_type: string; // "2-5-1", "passing_diminished", etc.
  start_time: number;
  end_time: number;
  chords: string[];
  key: string;
  description: string;  // Human-readable explanation
}

// Full Result
interface TranscriptionResult {
  notes: NoteEvent[];
  chords: ChordEvent[];
  tempo?: number;
  time_signature?: string;
  key?: string;
  duration: number;
  patterns: GospelPattern[];
  difficulty?: "beginner" | "intermediate" | "advanced";
  midi_url?: string;
  source_title?: string;
}

// Job tracking
interface TranscriptionJob {
  id: string;
  status: JobStatus;
  progress: number;
  current_step?: string;
  source_url?: string;
  source_file?: string;
  options: TranscriptionOptions;
  result?: TranscriptionResult;
  error?: string;
  created_at: string;
  started_at?: string;
  completed_at?: string;
}

type JobStatus = 
  | "queued"
  | "downloading"
  | "processing"
  | "analyzing"
  | "complete"
  | "failed"
  | "cancelled";
```
